Using colord to convert LCH to RGB

// colord.ts
import { Plugin, colord, extend } from 'colord'
import lchPlugin from 'colord/plugins/lch'

extend([(lchPlugin as unknown) as Plugin])
export { colord }
colord({ h: hue, c: 72, l: 80 }).toRgbString()
